      document.addEventListener("DOMContentLoaded", function () {
        // Size selection functionality
        const sizeButtons = document.querySelectorAll("#size-buttons button");
        const sizeAvailability = document.getElementById("size-availability");
        const stockInfo = {
          S: 5,
          M: 8,
          L: 12,
          XL: 15,
          XXL: 3,
        };
        function updateSizeSelection(selectedButton) {
          sizeButtons.forEach((button) => {
            button.classList.remove("bg-primary", "text-white");
            button.classList.add("border-gray-600");
          });
          selectedButton.classList.remove("border-gray-600");
          selectedButton.classList.add("bg-primary", "text-white");
          const size = selectedButton.dataset.size;
          sizeAvailability.textContent = `${stockInfo[size]} articles disponibles`;
        }
        sizeButtons.forEach((button) => {
          button.addEventListener("click", () => updateSizeSelection(button));
        });
        // Initialize with size L selected
        const defaultSize = document.getElementById("size-L");
        updateSizeSelection(defaultSize);
        // Quantity input functionality
        const minusBtn = document.querySelector("button:has(.ri-subtract-line)");
        const plusBtn = document.querySelector("button:has(.ri-add-line)");
        const quantityInput = document.querySelector('input[type="number"]');
        minusBtn.addEventListener("click", function () {
          const currentValue = parseInt(quantityInput.value);
          if (currentValue > 1) {
            quantityInput.value = currentValue - 1;
          }
        });
        plusBtn.addEventListener("click", function () {
          const currentValue = parseInt(quantityInput.value);
          quantityInput.value = currentValue + 1;
        });
      });
      document.addEventListener("DOMContentLoaded", function () {
        // Product image gallery
        const thumbnails = document.querySelectorAll(".grid-cols-4 > div");
        const mainImage = document.querySelector(
          ".bg-gray-900.rounded-lg.p-4.mb-4 img",
        );
        thumbnails.forEach((thumbnail) => {
          thumbnail.addEventListener("click", function () {
            const thumbnailImg = this.querySelector("img");
            const thumbnailSrc = thumbnailImg.src;
            // Swap images
            const mainSrc = mainImage.src;
            mainImage.src = thumbnailSrc;
            thumbnailImg.src = mainSrc;
            // Update active state
            thumbnails.forEach((t) => t.classList.remove("border", "border-primary"));
            this.classList.add("border", "border-primary");
          });
        });
      });
